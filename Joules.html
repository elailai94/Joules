def capacitance_in_series(capacitances)
  total_capacitance = 0
  if capacitances.empty?
    return total_capacitance.to_f
  else
    capacitances.each do |capacitance|
      total_capacitance += (1.0 / capacitance)
    end
    return 1 / total_capacitance
  end
end
